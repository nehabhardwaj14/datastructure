           // convert into title case
              void titleCase(char *str)
             {
               int i=0;
               int capitalize=1;                  
               while(str[i])
               {
                 if((str[i]==0))
                 {
                   capitalize=1;
                 }
                 else if(capitalize)
                 {
                   str[i]=toupper(str[i]);
                   capitalize=0;
                 }
                 else
                 {
                   str[i]=tolower(str[i]);
                 }
                 i++;
               }
             }
