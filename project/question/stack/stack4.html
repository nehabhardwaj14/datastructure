          // dynamically stack operation on string
            struct Node{
              char str[25];
              struct node *next;
             };
              typedef struct Node node;
        
            int isempty(node *top)
              {
            if(top==NULL)
              {
                return 1;
               }
             return 0;
               }
             int isfull(node *ptr)
               {
             if(ptr==NULL)
               {
                  return 1;
                }
               return 0;
             }
             char *push(node **top,char val[])
            {
              node *nn;
              nn=(node *)malloc(sizeof(node));
              if(isfull(nn))
                {
                  return NULL;
                 }
               strcpy(nn->str,val);
               nn->next=*top;
               *top=nn;
               return &(nn->str);
              }
               int pop(node **top)
             {
               node *temp;
               if(isempty(*top))
                 {
                   return 0;
                  }
                  temp=*top;
                  (*top)=temp->next;
                  free(temp);
                  return 1;
               }
               char *peek(node *top)
             {
               if(isempty(top))
                 {
                   return NULL;
                  }
                  return &(top->str);
              }
