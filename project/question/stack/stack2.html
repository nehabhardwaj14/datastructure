           int isfull(int size,int top)
           {
             if(top==size-1){
               return 1;
               }
               return 0;
            }

             {
               if(top==-1)
            {
              return 1;
             }

           int pop(char stack[5][10],int top)
             {
               if(isempty(top))
             {
               return -1;
              }
            top--;
            return top;
              }

            int i;
